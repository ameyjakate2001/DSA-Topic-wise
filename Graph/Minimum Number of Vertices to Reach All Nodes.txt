     **Minimum Number of Vertices to Reach All Nodes
        => we cannot reach node that has indegree 0 from any other node, so we have to add them.
        => anyother node has indegree > 0 then we can reach that node from anyother node.
        => so we have to take that node who has indegree 0.


       vector<int> findSmallestSetOfVertices(int n, vector<vector<int>>& edges) {
        vector<int>v[n];
        int indegree[n];
        for(int i=0; i<n; i++) indegree[i]=0;
        for(int i=0; i<edges.size(); i++) {
            v[edges[i][0]].push_back(edges[i][1]);
            indegree[edges[i][1]]++;
        }
        vector<int>ans;
        for(int i=0; i<n; i++) 
            if(indegree[i] == 0) ans.push_back(i);
            return ans;
    }
